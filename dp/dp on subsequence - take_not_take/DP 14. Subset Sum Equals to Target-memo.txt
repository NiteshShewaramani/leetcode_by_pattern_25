class Solution {
    //TC O(target x n)
    //SC O(target x n)
    public boolean isSubsetSum(int[] arr, int target) {
        Boolean memo[][] = new Boolean[target+1][arr.length];
      return recursion(arr,target,arr.length-1,memo);
    }

    private boolean recursion(int[] arr , int target , int index,Boolean memo[][])
    {
        if(target==0)
            return true;

        if(index==0)
            return target == arr[0];


        if(memo[target][index]!=null)
            return memo[target][index];

        boolean notPick = recursion(arr,target,index-1,memo);
        boolean pick = false;

        if(target>=arr[index])
            pick = recursion(arr,target-arr[index],index-1,memo);

        return memo[target][index] = pick || notPick;            

    }
}
