class Solution {
    //TC O(N)
    //SC O(1)
    public int maxProfit(int[] prices) {
       int prevBuy =Integer.MAX_VALUE, profit=0;

       for(int i=0;i<prices.length;i++)
       {
        profit = Math.max(profit,prices[i]-prevBuy);

        prevBuy = Integer.min(prevBuy,prices[i]);

       } 
       return profit;
    }
}
