class Solution {
    public boolean isSubsequence(String s, String t) {
        int sp =0 , tP = 0 ;
        if(s.length()==0)
        return true;

        if(t.length()==0)
        return false;
        for(int i=0;i<t.length();i++)
        {
            if(s.charAt(sp) == t.charAt(i))
            {
                ++sp;
            }
            if(sp == s.length())
            return true;
        }

        if(sp == s.length())
            return true;

        return false;
    }
}
