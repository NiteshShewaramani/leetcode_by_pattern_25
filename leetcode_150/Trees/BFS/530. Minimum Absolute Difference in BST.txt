class Solution {
    int min=100000000;
    Integer prev = null;
    public int getMinimumDifference(TreeNode root) {
        dfs(root);
        return min;
    }
    public void dfs(TreeNode node)
    {
        if(node!=null)
        {
            dfs(node.left);
            if(prev!=null)
            {
                min = Math.min(min,node.val - prev);
            }
            prev = node.val;
            dfs(node.right);
        }
    }
}
