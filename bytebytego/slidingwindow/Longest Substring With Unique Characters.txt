import java.util.*;
public class Main {
    //TC O(N) SC O(1)
    public Integer longest_substring_with_unique_chars(String s) {
        int windowSize = 0 , right = 0 , left = 0 ;
        int lastSeen[] = new int[255];
            Arrays.fill(lastSeen,-1);
        //"cabcdeca"
        while(right < s.length()) {
            char c = s.charAt(right);
            //window broken element repeated in current window
            if(lastSeen[c] >= left) 
                left = lastSeen[c]+1;

            windowSize = Math.max(windowSize , (right-left)+1);    
            lastSeen[c] = right;
            right++;    
        }
        return windowSize;
    }
}
