public class Solution {
    public boolean hasCycle(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;

        while(slow!=null && fast.next != null)
        {
            fast = fast.next.next;
            slow = slow.next;
            if(fast==null || slow == null)
            return false;
            if(fast==slow)
            return true;
        }
        return false;
    }
}
