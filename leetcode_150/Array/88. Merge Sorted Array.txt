class Solution {
    //TC O(N+M)
    //SC O(1)
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int j=n-1;
        int k=m-1;
        for(int i=(m+n)-1;i>=0;i--)
        {
            if(k<0 || j<0) break;

            if(nums1[k]<nums2[j]) {
                nums1[i] = nums2[j--];
            }
            else {
                nums1[i]=nums1[k--];
            }

            
        }

        for(;j>=0;j--) {
            nums1[j] = nums2[j];
        }
        
    }
}
