class Solution {
    public ListNode removeNthFromEnd(ListNode head, int n) {
        ListNode fast = head;
        ListNode slow = head;

        if(head == null)
            return null;

        while(n>0 && fast != null) {
            fast=fast.next;
            --n;
        }
        if(fast == null)
            return head.next;
               
        
        while(fast.next!=null) {
            fast = fast.next;
            slow = slow.next;
        }
        slow.next = slow.next.next;

        return head;
    }
}
