class Solution {
    public boolean isAnagram(String s, String t) {
        int[] arr1 = new int[26];
        
        if(s.length() != t.length())
            return false;

         for(int i=0;i<s.length();i++) {
            int index = s.charAt(i)-97;
            arr1[index]++;
         }

         for(int i=0;i<t.length();i++) {
            int index = t.charAt(i)-97;
            arr1[index]--;
            if(arr1[index]<0)
                return false;
         }

        return true;   
        
    }
}
